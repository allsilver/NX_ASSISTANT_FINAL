# settings.jsonмқҳ db_mcp_token к°’мқ„ $tokм—җ
$tok  = (Get-Content .\server\config\settings.json -Raw | ConvertFrom-Json).db_mcp_token
$body = @{ question="лҢҗнҚј н”„лЎ нҠё м„Өкі„ кё°мӨҖ м•Ңл ӨмӨҳ"; domain="<мӢӨм ң_лҸ„л©”мқёнӮӨ>"; for_gpt=$false } | ConvertTo-Json
$r = Invoke-RestMethod -Uri "http://127.0.0.1:8766/mech/ask" -Method Post `
      -Headers @{Authorization="Bearer $tok"} -ContentType "application/json" -Body $body
"answer кёёмқҙ: $($r.answer.Length)"
"мқҙлҜём§Җ мҲҳ: $($r.images.Count)"
$r.images | ForEach-Object { "  [$($_.score_pct)%] $($_.name)  (base64 $($_.data.Length)мһҗ)" }
